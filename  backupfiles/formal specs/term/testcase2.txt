// A program that accepts a single natural number n and writes out the sequence 1 2 3 4 ... n
//Required: n >= 1
//Alg: 
//	repeat until n == 0
//		push n
//		n--
//	for i = 1 to n
//		write

TC 0				//Push 0 as a first value on the stack to simplify cycle (*)
TA 2   				//save n to address 2, push address 1 and n
RD	
ST
TA 1
TA 2
DR

LAB "start" 		
ST					//load the current counter value,
TA 1				
DR

TA 1				//store decremented counter in address 1
TA 1
DR
TC -1
AD

JNZ "start"			//repeat until current counter == 0
ST		

LAB "start2"		//a second for cycle (*) that writes all the values on the stack
WR
JNZ "start2"

//Full program:
TC 0 ; TA 2 ; RD ; ST ; TA 1 ; TA 2 ; DR ; LAB "start" ; ST ; TA 1 ; DR ; TA 1 ; TA 1 ; DR ; TC -1 ; AD ; JNZ "start" ; ST ; LAB "start2" ; WR ; JNZ "start2"
